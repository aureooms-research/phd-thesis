\begin{problem}[GPT on the real line]
	Given \(n\) points \(q_1, \ldots, q_n \in \mathbb{R}\), decide
	whether there exist \(i < j \in [n]\) such that
	\begin{displaymath}
		\det
		\left(
		\begin{matrix}
		1 & q_i \\
		1 & q_j
		\end{matrix}
		\right)
		= q_j - q_i
		= 0.
	\end{displaymath}
\end{problem}
